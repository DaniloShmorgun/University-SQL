use [Облік виданих позик]
go

select ПІП_клієнта, позика_погашена
from Клієнти, Повернення_позик
where Клієнти.код_клієнта = Повернення_позик.код_клієнта
and (Повернення_позик.позика_погашена = 0 and cast(getdate() as date) > Клієнти.дата_закінчення_договору)

select avg(розмір_позики) [Середній розмір позики]
from Клієнти

select ПІП_клієнта [Людина з найбільшою позикою]
from Клієнти кл1
inner join 
 (select max(розмір_позики) [Максимальний_розмір_позики] from Клієнти) frr
 on кл1.розмір_позики = frr.Максимальний_розмір_позики
